"CREATE TABLE WINDOW(
ID int PRIMARY KEY,
WIDTH int,
HEIGHT int,
FPS int,
R int,
G int,
B int,
A int,
TITLE varchar(255),
FULLSCREEN int,
IMGUI_INI varchar(255)
)";

"CREATE TABLE TILEMAP(
ID int PRIMARY KEY,
TEXTURE_ID int NOT NULL,
POS_X decimal NOT NULL,
POS_Y decimal NOT NULL,
TILES_SIZE int NOT NULL,
FOREIGN KEY(TEXTURE_ID) REFERENCES TEXTURES(ID)
)";

"CREATE TABLE TILES(
ID int,
TILEMAP_ID int,
POS_X decimal NOT NULL,
POS_Y decimal NOT NULL,
SIZE_X decimal NOT NULL,
SIZE_Y decimal NOT NULL,
WALKABLE int NOT NULL,
ITEM_ID int NOT NULL,
CHARACTER_ID int NOT NULL,
PRIMARY KEY(ID, TILEMAP_ID),
FOREIGN KEY(TILEMAP_ID) REFERENCES TILEMAP(ID)
)";

"CREATE TABLE TEXTURES(
ID INTEGER PRIMARY KEY AUTOINCREMENT,
FILE_DIR varchar(255),
TAG varchar(255),
UNIQUE(FILE_DIR, TAG)
)";

"CREATE TABLE FONTS(
ID INTEGER PRIMARY KEY AUTOINCREMENT,
FILE_DIR varchar(255),
TAG varchar(255),
UNIQUE(FILE_DIR, TAG)
)";

"CREATE TABLE ITEMS(
ID INTEGER PRIMARY KEY AUTOINCREMENT,
NAME varchar(255) NOT NULL,
HEALTH varchar(255) NOT NULL,
DEFENSE int NOT NULL,
ATTACK int NOT NULL,
MANA int NOT NULL,
CRITTIK_ATTACK decimal NOT NULL,
LEGENDARY_ITEM int NOT NULL,
DROPPED_ITEM INT NOT NULL,
ITEM_COINS INT NOT NULL
)";

"CREATE TABLE CHARACTERS(
ID INTEGER PRIMARY KEY AUTOINCREMENT,
TEXTURE_ID int NOT NULL,
POS_X decimal NOT NULL,
POS_Y decimal NOT NULL,
SCALE_X decimal NOT NULL,
SCALE_Y decimal NOT NULL,
HEALTH decimal NOT NULL,
MANA decimal NOT NULL,
ATTACK decimal NOT NULL,
DEFENSE decimal NOT NULL,
CRIT_ATTACK decimal NOT NULL,
TURNS int NOT NULL,
LEVEL int NOT NULL,
ALIVE int NOT NULL,
EXP int NOT NULL,
CHARTYPE int NOT NULL,
RANGE int NOT NULL,
COL int NOT NULL,
NAME varchar NOT NULL,
MAX_EXPERIENCE_ int NOT NULL,
FOREIGN KEY(TEXTURE_ID) REFERENCES TEXTURES(ID)
)";

"CREATE TABLE CLASSES(
ID INTEGER PRIMARY KEY AUTOINCREMENT,
PROFESSION varchar(255) NOT NULL,
HEALTH int NOT NULL,
MANA int NOT NULL,
ATTACK int NOT NULL,
DEFENSE int NOT NULL,
CRIT_ATTACK int NOT NULL,
DESCRIPTION varchar(255) NOT NULL,
UNIQUE(PROFESSION, HEALTH, MANA,
ATTACK, DEFENSE, CRIT_ATTACK,
DESCRIPTION)
)";
